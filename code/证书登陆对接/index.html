KCSP-ADMIN 证书登陆

kcsp-admin 应用系统对接证书登陆,对接流程

[toc]

前置流程

第三方应用整合证书前置流程:

  • 安装证书中间件
  • 准备好证书链文件以及证书文件
  • 前端需要导入和中间件通信js,gen-js

对接概述

大致可以分为以下三个部分

  • 前端整合gen-js ,和证书中间件进行交互，此时涉及到从后端获取随机数
  • 后端提供登陆接口随机数获取接口
  • 后端整合登陆接口

  • 模板页导入gen-js

export default {
  data() {
    return {
      certs: [],
      certPing: '',
      pkiLoginInfo: {},
      koalPkiServerInfo: {
        SERVICE_TYPE: 'http://',
        SERVER_IP: '127.0.0.1',
        SERVER_PORT: '18080',
      },
      url: {
        querySeed: '/sys/certrandom',
      },
 
    }
  },
  methods: {
    loginUseCert() {
      let cert = this.certs[2]
      console.log(cert.hasPIN)
      if (cert.hasPIN === '0') {
        //没有 pin
        this.loginUserNoPinCert(cert)
      } else {
        //有pin
        this.loginWithPin(cert)
      }
    },
    handleTabClick(key) {
      this.customActiveKey = key
      // this.form.resetFields()
    },
    loginWithPin(cert) {
      if (this.certPing) {
        this.$message.warning('请输入ping码')
      }
      // ping 一起传进来
      KoalService.verifyPIN(
        this.koalPkiServerInfo,
        this.pkiLoginInfo,
        cert,
        this.certPing,
        () => {
          //cert认证后可以修改
          this.loginUserNoPinCert(cert)
        },
        (error) => {
          this.$message.warning(error)
        }
      )
    },
    loginUserNoPinCert(cert) {
      let cn = cert.subjectName.CN
      postAction(this.url.querySeed, {
        cn: cn,
      }).then((res) => {
        if (res.success) {
          let singnB64 = res.result
          console.log(singnB64)
          KoalService.signdata(
            this.koalPkiServerInfo,
            this.pkiLoginInfo,
            cert,
            singnB64,
            (singDataStr) => {
              this.callLogin(cert, singDataStr)
            },
            (error) => {
              this.$message.warning(error)
            }
          )
        } else {
          this.$message.warning(res.message)
        }
      })
    },
    callLogin(cert, singDataStr) {
      KoalService.exportCertificate(this.koalPkiServerInfo, this.pkiLoginInfo, cert, (certB64) => {
        let params = {
          signData: singDataStr,
          certB64: certB64,
        }
        this.CertLogin(params)
          .then((res) => {
            this.$refs.loginSelect.show(res.result)
          })
          .catch((err) => {
            this.$message.warning(err.message)
          })
      })
    },
    chooseCert() {
      KoalService.pkiLogin(
        this.koalPkiServerInfo,
        (loginInfo) => {
          this.pkiLoginInfo = loginInfo
          this.showChooseCert()
        },
        (error) => {
          this.$message.warning(error.message)
        }
      )
    },
    showChooseCert() {
      KoalService.getAllCert(
        this.koalPkiServerInfo,
        (certs) => {
          this.certs = certs
          console.log(this.certs)
        },
        (error) => {
          this.$message.warning(error)
        },
        this.pkiLoginInfo
      )
    },
    loginSelectOk() {
      this.loginSuccess()
    },
    loginSuccess() {
      // update-begin- author:sunjianlei --- date:20190812 --- for: 登录成功后不解除禁用按钮，防止多次点击
      // this.loginBtn = false
      // update-end- author:sunjianlei --- date:20190812 --- for: 登录成功后不解除禁用按钮，防止多次点击
      this.$router.push({ path: '/dashboard/analysis' }).catch(() => {
        console.log('登录跳转首页出错,这个错误从哪里来的')
      })
      this.$notification.success({
        message: '欢迎',
        description: `${timeFix()}，欢迎回来`,
      })
    },
    cmsFailed(err) {
      this.$notification['error']({
        message: '登录失败',
        description: err,
        duration: 4,
      })
    },
  },
}

        KoalCertB64 param = new KoalCertB64();
        BeanUtils.copyProperties(sysLoginModel, param);
        return login(param);
    }

    public Result<JSONObject> login(KoalCertB64 param) throws Exception {
        String certB64 = param.certB64;
        Certificate cert = CertUtils.parseCert(certB64);
        Function<String,String> plainSignB64Apply = new Function<String, String>() {
            @Override
            public String apply(String cn) {
                String plainBase64 = (String) cache.shget(RedisKeyConstant.KOAL_CERT_LOGINHASH, cn);
                cache.shdel(RedisKeyConstant.KOAL_CERT_LOGINHASH, cn);
                return plainBase64;
            }
        };

        // 证书链认证先跳过
        // 所谓loginkey 其实对应的是用户的username
//        String loginKey = "admin";
        String loginKey = checkCertAndReturnLoginKey(param.signData, cert, plainSignB64Apply);

        SysUser sysuser = sysUserService.getUserByName(loginKey);
        if (sysuser == null) {
            sysuser = sysUserService.getUserByCertLogin(loginKey);
        }
        Result<JSONObject> result = new Result();
        result.error500("未找到用户");
        if (sysuser == null) {
            return result;
        }else {
            userInfo(sysuser, result);
        }
        return result;
    }

    /**
     * 用户信息
     *
     * @param sysUser
     * @param result
     * @return
     */
    private Result<JSONObject> userInfo(SysUser sysUser, Result<JSONObject> result) {
        String syspassword = sysUser.getPassword();
        String username = sysUser.getUsername();
        // 生成token
        String token = JwtUtil.sign(username, syspassword);
        // 设置token缓存有效时间
        cache.set(CommonConstant.PREFIX_USER_TOKEN + token, token);
        cache.expire(CommonConstant.PREFIX_USER_TOKEN + token, JwtUtil.EXPIRE_TIME*2 / 1000);

        // 获取用户部门信息
        JSONObject obj = new JSONObject();
        List<SysDepart> departs = sysDepartService.queryUserDeparts(sysUser.getId());
        obj.put("departs", departs);
        if (departs == null || departs.size() == 0) {
            obj.put("multi_depart", 0);
        } else if (departs.size() == 1) {
            sysUserService.updateUserDepart(username, departs.get(0).getOrgCode());
            obj.put("multi_depart", 1);
        } else {
            //查询当前是否有登录部门
            // update-begin--Author:wangshuai Date:20200805 for：如果用戶为选择部门，数据库为存在上一次登录部门，则取一条存进去
            SysUser sysUserById = sysUserService.getById(sysUser.getId());
            if(oConvertUtils.isEmpty(sysUserById.getOrgCode())){
                sysUserService.updateUserDepart(username, departs.get(0).getOrgCode());
            }
            // update-end--Author:wangshuai Date:20200805 for：如果用戶为选择部门，数据库为存在上一次登录部门，则取一条存进去
            obj.put("multi_depart", 2);
        }
        obj.put("token", token);
        obj.put("userInfo", sysUser);
        obj.put("sysAllDictItems", sysDictService.queryAllDictItems());
        result.setResult(obj);
        result.success("登录成功");
        return result;
    }


    /**
     * 检查登录有效性以及获取登录证书中的身份标识
     * @param signDateB64
     * @param cert
     * @param plainSignB64Apply
     * @return
     * @throws Exception
     */
    public  String checkCertAndReturnLoginKey(String signDateB64, Certificate cert, Function<String, String> plainSignB64Apply) throws Exception {
        verifyCert(cert);
        String loginKey = getLoginKeyOfCertDn(cert);
        String loginCn = cert.getSubjectCommonName();
        String plainSignB64 = plainSignB64Apply.apply(loginCn);
        if (plainSignB64 == null) {
            log.error("error get plainSignB64 {}",signDateB64);
            throw new Exception("系统错误,未获取签名随机数");
        }
        if (!SignVerifyUtil.verifyFixSignAlgoSignature(cert, signDateB64, Base64.decodeBase64(plainSignB64))) {
            log.error("error verifySignature signDateB64 {}",signDateB64);
            throw new Exception("验证随机数签名失败");
        }
        return loginKey;
    }

    /**
     * 获取用以登录的证书标识
     * @param cert
     * @return
     */
    private  String getLoginKeyOfCertDn(Certificate cert) throws Exception {
        Map<String, String> dnMap = CertUtils.getDnMapFromCert(cert);
        String certLoginKey = dnMap.get(LOGIN_KEY_OF_DN);
        if (certLoginKey == null) {
            throw new Exception(String.format("证书DN中未找到 %s 选项", LOGIN_KEY_OF_DN));
        }
        return certLoginKey;
    }

    public void verifyCert(Certificate cert) throws Exception {
        if (!isValidate(cert)) {
            log.error("InValidate Cert, Cert {}", cert.getSubject().toString());
            throw new Exception("证书不在有效期内");
        }

        String trustChainCertB64 = (String) cache.shget(RedisKeyConstant.KOAL_CERT_CHAIN, cert.getIssuer().toString());

        if (trustChainCertB64 == null) {
            log.error("has not trust chain");
            throw new Exception("无可用验证客户端证书的证书链");
        }

        Certificate trustCert = CertUtils.parseCert(trustChainCertB64);
        try {
            if (!CertChains.verifyCerts(trustCert, cert)) {
                log.error("cloud not trust client cert Cert {}", cert.getSubject().toString());
                throw new Exception("使用客户端证书链验证登录证书失败,无法信任登录证书");
            }
        } catch (Exception e) {
            log.error("unKnow exception when trust client cert Cert {}",  cert.getSubject().toString());
            throw new Exception("使用客户端证书链验证登录证书异常 :" + e.getMessage());
        }

    }
    /**
     * 是否已过有效期
     * @param cert
     * @return
     */
    public static boolean isValidate(Certificate cert) {
        Date tStart = cert.getValidFromValue();
        Date tEnd = cert.getValidToValue();
        Calendar c = Calendar.getInstance();
        return c.getTime().compareTo(tStart) >= 0 && c.getTime().compareTo(tEnd) <= 0;
    }
