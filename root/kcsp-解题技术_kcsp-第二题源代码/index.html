fun getRect(x: Int, y: Int):Rect {
    val x_step = 32 // 886 就是1
    val y_step = 3
    return Rect(
        x + x_step + 2, y,
        Point(x, y), Point(x, y + y_step), Point(x + x_step, y), Point(x + x_step, y + y_step))
}

fun colorToRGB(red: Int, green: Int, blue: Int): Int {
    var newPixel = 0
    newPixel = newPixel shl 8
    newPixel += red
    newPixel = newPixel shl 8
    newPixel += green
    newPixel shl 8
    newPixel += blue
    return newPixel
}

fun scanQR(fp: String) {
    try {
        val formatReader = MultiFormatReader()
        //读取指定的二维码文件
        //读取指定的二维码文件
        val bufferedImage: BufferedImage = ImageIO.read(File(fp))
        val binaryBitmap = BinaryBitmap(HybridBinarizer(BufferedImageLuminanceSource(bufferedImage)))
        //定义二维码参数
        //定义二维码参数
//    val hints = HashMap<Any, Any>()
//    hints[EncodeHintType.CHARACTER_SET] = "utf-8"
        val result: com.google.zxing.Result = formatReader.decode(binaryBitmap)
        //输出相关的二维码信息
        //输出相关的二维码信息
        println("文件路径: $fp")
        println("解析结果：" + result.toString())
        println("二维码格式类型：" + result.getBarcodeFormat())
        println("二维码文本内容：" + result.getText())
    } catch (throwable: Throwable) {
        println("$fp scan error")
    }
}
